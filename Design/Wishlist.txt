CREATE TABLE `Wishlist` (
	id		INT(60) NOT NULL auto_increment,
	user_id		VARCHAR(40) NOT NULL,
	product_id	INT(60) NOT NULL,

	PRIMARY KEY(id),
	FOREIGN KEY(user_id) REFERENCES `Users`(UserID)
		ON DELETE CASCADE
		ON UPDATE CASCADE,
	FOREIGN KEY(product_id) REFERENCES `Products`(id)
		ON DELETE CASCADE
		ON UPDATE CASCADE
)
