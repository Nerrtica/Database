CREATE TABLE `Consumer` (
	UserID		VARCHAR(40) NOT NULL,
	Birth		DATE NULL,
	Sex		VARCHAR(5) NOT NULL
				CHECK(VALUE IN('MAN', 'WOMAN')),	
	OwnCoupon	INTEGER(20) NULL,

	PRIMARY KEY(UserID),
	FOREIGN KEY(UserID)
		REFERENCES `Users`(UserID)
		ON DELETE RESTRICT
		ON UPDATE RESTRICT,
	FOREIGN KEY(OwnCoupon)
		REFERENCES `Coupon`(ID)
		ON DELETE SET NULL
		ON UPDATE RESTRICT
);
