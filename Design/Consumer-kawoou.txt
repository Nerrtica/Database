CREATE DATABASES `Consumer` (
	UserID		VARCHAR(40) NOT NULL,
	Birth		DATE NULL,
	Sex		INTEGER(2) NOT NULL CONSTRAINT `Sex-Info`
				CHECK(VALUE IN('MAN', 'WOMAN')),	
	OwnCoupon	INTEGER(20) NULL,

	PRIMARY KEY(UserID),
	FOREIGN KEY(UserID)
		REFERENCES `Users`(UserID)
		ON DELETE CASCADE
		ON UPDATE CASCADE,
	FOREIGN KEY(OwnCoupon)
		REFERENCES `Coupon`(ID)
		ON DELETE SET NULL
		ON UPDATE CASCADE
)
